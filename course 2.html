Practice Problems

Python: Reverse a List

Easy

Write a Python function that reverses a list without using the built-in reverse() method.

def reverse_list(lst):
    return lst[::-1]

# Alternative solution
def reverse_list_manual(lst):
    return [lst[i] for i in range(len(lst)-1, -1, -1)]

Java: Two Sum

Medium

Given an array of integers, return indices of the two numbers such that they add up to a target.

public int[] twoSum(int[] nums, int target) {
    Map<Integer,Integer> map = new HashMap<>();
    for(int i=0; i<nums.length; i++){
        int complement = target - nums[i];
        if(map.containsKey(complement)) {
            return new int[]{map.get(complement), i};
        }
        map.put(nums[i], i);
    }
    return new int[]{};
}

JavaScript: Factorial

Easy

Write a JS function to calculate factorial of a number recursively.

function factorial(n){
    if(n === 0 || n === 1) return 1;
    return n * factorial(n - 1);
}

// Iterative solution
function factorialIterative(n) {
    let result = 1;
    for(let i = 2; i <= n; i++) {
        result *= i;
    }
    return result;
}

C++: Palindrome Check

Medium

Check if a given string is a palindrome in C++ without using extra space.

#include <string>
bool isPalindrome(const std::string &s){
    int left = 0, right = s.size() - 1;
    while(left < right){
        if(s[left++] != s[right--]) {
            return false;
        }
    }
    return true;
}

Rust: Fibonacci

Medium

Write a function in Rust to return nth Fibonacci number using recursion.

fn fib(n: u32) -> u32 {
    if n == 0 || n == 1 { 
        return n; 
    }
    fib(n - 1) + fib(n - 2)
}

// Optimized iterative solution
fn fib_iterative(n: u32) -> u32 {
    let (mut a, mut b) = (0, 1);
    for _ in 0..n {
        let temp = a;
        a = b;
        b = temp + b;
    }
    a
}
